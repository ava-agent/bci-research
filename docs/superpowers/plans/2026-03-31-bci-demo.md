# BCI Agent Demo Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an interactive web demo where visitors see real-time synthetic brainwave visualizations and an AI agent that responds to brain state changes, deployed on Vercel (frontend) + Cloudbase cloud function (backend).

**Architecture:** Frontend generates synthetic EEG in JavaScript, runs FFT band extraction and state classification client-side, and renders a dashboard (ECharts waveform + band bars + status cards + AI chat). When brain state changes or user sends a message, frontend POSTs to a Cloudbase Python cloud function that invokes GLM-4-Flash via LangGraph and returns the response.

**Tech Stack:** Next.js 14 (App Router), TailwindCSS, ECharts, TypeScript (frontend); Python, Cloudbase cloud function, LangGraph, langchain-openai (backend)

### Task 2: Implement cloud function handler

**Files:**
- Create: `cloud/bci_agent/index.py`

The cloud function receives brain state + optional user message, invokes GLM-4-Flash via LangGraph, returns the agent response. This reuses the agent logic from `app/src/bci_agent/agent.py` but simplified for serverless (no BrainFlow dependency).

- [ ] **Step 1: Write index.py**

```python
"""Cloudbase cloud function: BCI Agent endpoint."""
import json
import os
from typing import Literal, TypedDict

from langchain_openai import ChatOpenAI
from langchain_core.messages import BaseMessage, HumanMessage, SystemMessage
from langchain_core.tools import tool
from langgraph.graph import END, StateGraph
from langgraph.prebuilt import ToolNode

# ── Tools ──

@tool
def suggest_activity(brain_state: str) -> str:
    """基于当前脑状态推荐合适的活动。

    Args:
        brain_state: 当前脑状态 (focused/relaxed/fatigued/command)
    """
    suggestions = {
        "focused": "专注度高，适合处理复杂任务：编程、写作、学习新概念",
        "relaxed": "状态放松，适合：创意思考、头脑风暴、轻松阅读",
        "fatigued": "检测到疲劳，建议：休息5分钟、深呼吸、喝水、站起来走动",
        "command": "已准备好接收指令，请下达命令",
    }
    return suggestions.get(brain_state, "继续保持当前状态")

TOOLS = [suggest_activity]

# ── Intent text mapping ──

_INTENT_TEXT = {
    "focused": "用户进入高度专注状态，可能正在集中注意力思考某个问题",
    "relaxed": "用户处于放松状态，可能在休息或冥想",
    "fatigued": "检测到认知疲劳信号，用户可能需要休息",
    "command": "检测到意图命令信号，用户想要执行某个操作",
    "idle": "用户处于空闲状态",
}

# ── LangGraph Agent ──

SYSTEM_PROMPT = """\
你是一个脑机接口 AI Agent。你通过 BCI 设备接收用户的脑电信号，\
将解码后的意图转化为有用的响应和行动。

当前脑状态: {brain_state}
置信度: {confidence:.0%}
频带能量: {band_powers}

根据用户的脑状态和解码意图，提供有帮助的中文响应。\
你可以使用工具来执行操作。保持回复简洁。"""


class AgentState(TypedDict):
    brain_state: str
    confidence: float
    intent_text: str
    band_powers: dict
    messages: list[BaseMessage]
    response: str


def interpret_intent(state: AgentState) -> dict:
    system = SYSTEM_PROMPT.format(
        brain_state=state["brain_state"],
        confidence=state["confidence"],
        band_powers=state["band_powers"],
    )
    user_content = f"[BCI 意图解码] {state['intent_text']}"
    if state.get("user_message"):
        user_content += f"\n\n用户消息: {state['user_message']}"
    messages = [
        SystemMessage(content=system),
        HumanMessage(content=user_content),
    ]
    return {"messages": messages}


def call_llm(state: AgentState) -> dict:
    llm = ChatOpenAI(
        model="glm-4-flash",
        base_url="https://open.bigmodel.cn/api/paas/v4/",
        api_key=os.environ.get("GLM_API_KEY", ""),
    ).bind_tools(TOOLS)
    response = llm.invoke(state["messages"])
    return {"messages": state["messages"] + [response]}


def format_response(state: AgentState) -> dict:
    last = state["messages"][-1]
    text = last.content if isinstance(last.content, str) else str(last.content)
    return {"response": text}


def _route_after_llm(state: AgentState) -> Literal["tools", "respond"]:
    last = state["messages"][-1]
    if hasattr(last, "tool_calls") and last.tool_calls:
        return "tools"
    return "respond"


def _build_graph():
    tool_node = ToolNode(TOOLS)
    g = StateGraph(AgentState)
    g.add_node("interpret", interpret_intent)
    g.add_node("llm", call_llm)
    g.add_node("tools", tool_node)
    g.add_node("respond", format_response)
    g.set_entry_point("interpret")
    g.add_edge("interpret", "llm")
    g.add_conditional_edges("llm", _route_after_llm, {
        "tools": "tools",
        "respond": "respond",
    })
    g.add_edge("tools", "llm")
    g.add_edge("respond", END)
    return g.compile()

_graph = _build_graph()

# ── Cloudbase entry point ──

def main(event, context):
    """Cloudbase cloud function handler."""
    # Handle CORS preflight
    if event.get("httpMethod") == "OPTIONS":
        return {
            "statusCode": 200,
            "headers": {
                "Access-Control-Allow-Origin": "*",
                "Access-Control-Allow-Methods": "POST, OPTIONS",
                "Access-Control-Allow-Headers": "Content-Type",
            },
            "body": "",
        }

    try:
        body = json.loads(event.get("body", "{}"))
    except (json.JSONDecodeError, TypeError):
        body = {}

    state = body.get("state", "idle")
    confidence = body.get("confidence", 0.5)
    bands = body.get("bands", {})
    user_message = body.get("message", "")

    intent_text = _INTENT_TEXT.get(state, "未知状态")
    if user_message:
        intent_text += f"。用户说: {user_message}"

    result = _graph.invoke({
        "brain_state": state,
        "confidence": confidence,
        "intent_text": intent_text,
        "band_powers": bands,
        "messages": [],
        "response": "",
        "user_message": user_message,
    })

    return {
        "statusCode": 200,
        "headers": {
            "Access-Control-Allow-Origin": "*",
            "Content-Type": "application/json",
        },
        "body": json.dumps({
            "response": result["response"],
        }, ensure_ascii=False),
    }
```

## Chunk 3: Frontend — Signal Engine (Core Logic)

### Task 5: Synthetic EEG signal generator

**Files:**
- Create: `web/lib/signal-generator.ts`

Generates synthetic multi-channel EEG by summing sinusoidal components at δ(2Hz), θ(6Hz), α(10Hz), β(20Hz), γ(40Hz) frequencies with configurable amplitudes plus Gaussian noise. Supports preset "brain states" that weight the amplitudes differently.

- [ ] **Step 1: Write signal-generator.ts**

```typescript
export type BrainState = "focused" | "relaxed" | "fatigued" | "command" | "idle";

export interface SignalPoint {
  timestamp: number;
  channels: number[];
}

// Band frequency centers (Hz)
const BANDS = {
  delta: 2,
  theta: 6,
  alpha: 10,
  beta: 20,
  gamma: 40,
} as const;

// Amplitude profiles per brain state
const STATE_PROFILES: Record<BrainState, Record<string, number>> = {
  focused:  { delta: 0.1, theta: 0.1, alpha: 0.2, beta: 0.8, gamma: 0.5 },
  relaxed:  { delta: 0.2, theta: 0.3, alpha: 0.8, beta: 0.2, gamma: 0.1 },
  fatigued: { delta: 0.3, theta: 0.7, alpha: 0.4, beta: 0.2, gamma: 0.1 },
  command:  { delta: 0.1, theta: 0.2, alpha: 0.3, beta: 0.7, gamma: 0.6 },
  idle:     { delta: 0.3, theta: 0.3, alpha: 0.3, beta: 0.3, gamma: 0.2 },
};

const SAMPLING_RATE = 250; // Hz
const NUM_CHANNELS = 4;

export class SignalGenerator {
  private sampleIndex = 0;
  private state: BrainState = "idle";
  private targetProfile: Record<string, number>;
  private currentProfile: Record<string, number>;

  constructor() {
    this.targetProfile = { ...STATE_PROFILES.idle };
    this.currentProfile = { ...STATE_PROFILES.idle };
  }

  setState(state: BrainState) {
    this.state = state;
    this.targetProfile = { ...STATE_PROFILES[state] };
  }

  getState(): BrainState {
    return this.state;
  }

  /** Generate next batch of samples */
  generate(numSamples: number): SignalPoint[] {
    const points: SignalPoint[] = [];

    for (let i = 0; i < numSamples; i++) {
      // Smooth transition toward target profile
      for (const band of Object.keys(this.currentProfile)) {
        this.currentProfile[band] +=
          (this.targetProfile[band] - this.currentProfile[band]) * 0.01;
      }

      const t = this.sampleIndex / SAMPLING_RATE;
      const channels: number[] = [];

      for (let ch = 0; ch < NUM_CHANNELS; ch++) {
        let value = 0;
        const phaseOffset = ch * 0.5;

        for (const [band, freq] of Object.entries(BANDS)) {
          const amplitude = this.currentProfile[band] ?? 0.1;
          value += amplitude * Math.sin(2 * Math.PI * freq * t + phaseOffset);
        }

        // Add Gaussian noise
        value += (Math.random() + Math.random() + Math.random() - 1.5) * 0.3;
        channels.push(value);
      }

      points.push({ timestamp: t, channels });
      this.sampleIndex++;
    }

    return points;
  }
}

export { SAMPLING_RATE, NUM_CHANNELS, BANDS };
```

### Task 6: FFT band power extraction + state decoder

**Files:**
- Create: `web/lib/band-analyzer.ts`

Implements a simple DFT-based band power extraction (no external FFT library needed for the small window sizes we use) and the state classifier ported from Python `decoder.py`.

- [ ] **Step 1: Write band-analyzer.ts**

```typescript
import type { SignalPoint } from "./signal-generator";
import { SAMPLING_RATE } from "./signal-generator";

export interface BandPowers {
  delta: number;
  theta: number;
  alpha: number;
  beta: number;
  gamma: number;
}

export interface BrainDecoded {
  state: string;
  confidence: number;
  intent: string;
  bands: BandPowers;
}

// Band frequency ranges (Hz)
const BAND_RANGES: [string, number, number][] = [
  ["delta", 1, 4],
  ["theta", 4, 8],
  ["alpha", 8, 13],
  ["beta", 13, 30],
  ["gamma", 30, 45],
];

/** Compute power spectral density using DFT for a specific frequency band */
function bandPower(signal: number[], sampleRate: number, fLow: number, fHigh: number): number {
  const n = signal.length;
  let power = 0;
  const freqStep = sampleRate / n;

  const kLow = Math.ceil(fLow / freqStep);
  const kHigh = Math.floor(fHigh / freqStep);

  for (let k = kLow; k <= kHigh && k < n / 2; k++) {
    let re = 0, im = 0;
    for (let i = 0; i < n; i++) {
      const angle = (2 * Math.PI * k * i) / n;
      re += signal[i] * Math.cos(angle);
      im -= signal[i] * Math.sin(angle);
    }
    power += (re * re + im * im) / (n * n);
  }

  return power;
}

/** Extract band powers from a window of signal points (averages across channels) */
export function extractBandPowers(points: SignalPoint[]): BandPowers {
  const numChannels = points[0]?.channels.length ?? 1;
  const bands: BandPowers = { delta: 0, theta: 0, alpha: 0, beta: 0, gamma: 0 };

  for (let ch = 0; ch < numChannels; ch++) {
    const signal = points.map((p) => p.channels[ch]);
    for (const [name, fLow, fHigh] of BAND_RANGES) {
      bands[name as keyof BandPowers] += bandPower(signal, SAMPLING_RATE, fLow, fHigh);
    }
  }

  // Average across channels
  for (const key of Object.keys(bands) as (keyof BandPowers)[]) {
    bands[key] /= numChannels;
  }

  // Normalize to sum to 1
  const total = Object.values(bands).reduce((a, b) => a + b, 0) || 1;
  for (const key of Object.keys(bands) as (keyof BandPowers)[]) {
    bands[key] /= total;
  }

  return bands;
}

const INTENT_TEXT: Record<string, string> = {
  focused: "用户进入高度专注状态，可能正在集中注意力思考某个问题",
  relaxed: "用户处于放松状态，可能在休息或冥想",
  fatigued: "检测到认知疲劳信号，用户可能需要休息",
  command: "检测到意图命令信号，用户想要执行某个操作",
  idle: "用户处于空闲状态",
};

/** Classify brain state from band powers (ported from decoder.py) */
export function classifyState(bands: BandPowers): BrainDecoded {
  const alpha = Math.max(bands.alpha, 1e-6);
  const beta = Math.max(bands.beta, 1e-6);
  const theta = Math.max(bands.theta, 1e-6);
  const gamma = bands.gamma;

  const ratioBetaAlpha = beta / alpha;
  const ratioAlphaBeta = alpha / beta;
  const ratioThetaAlpha = theta / alpha;

  let state: string;
  let confidence: number;

  if (ratioBetaAlpha > 2.0) {
    state = "focused";
    confidence = Math.min(ratioBetaAlpha / 4.0, 1.0);
  } else if (ratioAlphaBeta > 2.0) {
    state = "relaxed";
    confidence = Math.min(ratioAlphaBeta / 4.0, 1.0);
  } else if (ratioThetaAlpha > 1.5) {
    state = "fatigued";
    confidence = Math.min(ratioThetaAlpha / 3.0, 1.0);
  } else if (beta > 0.3 && gamma > 0.2) {
    state = "command";
    confidence = 0.6;
  } else {
    state = "idle";
    confidence = 0.3;
  }

  return {
    state,
    confidence,
    intent: INTENT_TEXT[state] ?? "未知状态",
    bands,
  };
}
```

### Task 8: BCI engine hook (connects signal generator + analyzer)

**Files:**
- Create: `web/hooks/use-bci-engine.ts`

Custom React hook that runs the signal generator on a timer, extracts band powers, classifies state, and exposes the reactive data to components.

- [ ] **Step 1: Write use-bci-engine.ts**

```typescript
"use client";

import { useState, useRef, useCallback, useEffect } from "react";
import { SignalGenerator, SAMPLING_RATE } from "@/lib/signal-generator";
import type { SignalPoint, BrainState } from "@/lib/signal-generator";
import { extractBandPowers, classifyState } from "@/lib/band-analyzer";
import type { BandPowers, BrainDecoded } from "@/lib/band-analyzer";

const WINDOW_SECONDS = 10;
const MAX_POINTS = SAMPLING_RATE * WINDOW_SECONDS;
const GENERATE_INTERVAL_MS = 100;
const ANALYZE_INTERVAL_MS = 1000;
const SAMPLES_PER_TICK = Math.round(SAMPLING_RATE * (GENERATE_INTERVAL_MS / 1000));

export interface BciEngineState {
  waveform: SignalPoint[];
  bands: BandPowers;
  decoded: BrainDecoded;
  isRunning: boolean;
  start: () => void;
  stop: () => void;
  switchState: (state: BrainState) => void;
}

export function useBciEngine(): BciEngineState {
  const generatorRef = useRef(new SignalGenerator());
  const bufferRef = useRef<SignalPoint[]>([]);
  const genTimerRef = useRef<NodeJS.Timeout | null>(null);
  const analyzeTimerRef = useRef<NodeJS.Timeout | null>(null);

  const [waveform, setWaveform] = useState<SignalPoint[]>([]);
  const [bands, setBands] = useState<BandPowers>({
    delta: 0.2, theta: 0.2, alpha: 0.2, beta: 0.2, gamma: 0.2,
  });
  const [decoded, setDecoded] = useState<BrainDecoded>({
    state: "idle", confidence: 0, intent: "", bands: { delta: 0, theta: 0, alpha: 0, beta: 0, gamma: 0 },
  });
  const [isRunning, setIsRunning] = useState(false);

  const generate = useCallback(() => {
    const points = generatorRef.current.generate(SAMPLES_PER_TICK);
    bufferRef.current = [...bufferRef.current, ...points].slice(-MAX_POINTS);
    setWaveform([...bufferRef.current]);
  }, []);

  const analyze = useCallback(() => {
    const buf = bufferRef.current;
    if (buf.length < SAMPLING_RATE) return;
    const window = buf.slice(-SAMPLING_RATE); // Last 1 second
    const bandPowers = extractBandPowers(window);
    const result = classifyState(bandPowers);
    setBands(bandPowers);
    setDecoded(result);
  }, []);

  const start = useCallback(() => {
    if (genTimerRef.current) return;
    genTimerRef.current = setInterval(generate, GENERATE_INTERVAL_MS);
    analyzeTimerRef.current = setInterval(analyze, ANALYZE_INTERVAL_MS);
    setIsRunning(true);
  }, [generate, analyze]);

  const stop = useCallback(() => {
    if (genTimerRef.current) clearInterval(genTimerRef.current);
    if (analyzeTimerRef.current) clearInterval(analyzeTimerRef.current);
    genTimerRef.current = null;
    analyzeTimerRef.current = null;
    setIsRunning(false);
  }, []);

  const switchState = useCallback((state: BrainState) => {
    generatorRef.current.setState(state);
  }, []);

  useEffect(() => {
    return () => { stop(); };
  }, [stop]);

  return { waveform, bands, decoded, isRunning, start, stop, switchState };
}
```

```tsx
  return (
```

```tsx
"use client";

import { useState, useRef, useEffect } from "react";

export interface ChatMessage {
  id: string;
  type: "system" | "agent" | "user";
  text: string;
}

interface ChatPanelProps {
  messages: ChatMessage[];
  onSend: (text: string) => void;
  loading: boolean;
}

export function ChatPanel({ messages, onSend, loading }: ChatPanelProps) {

    setInput("");

  return (

              <div className="bg-indigo-50 text-indigo-600 rounded-lg px-3 py-2 text-sm">
                <span className="text-[10px] text-indigo-400 block mb-0.5">系统</span>
                {msg.text}
              </div>
            )}
            {msg.type === "agent" && (

                <span className="text-[10px] text-indigo-400 block mb-0.5">BCI Agent</span>

              </div>
            )}
            {msg.type === "user" && (
              <div className="bg-indigo-500 text-white rounded-lg px-3 py-2 text-sm">
                {msg.text}
              </div>
            )}
          </div>
        ))}
        {loading && (
          <div className="bg-slate-50 rounded-lg px-3 py-2 text-sm text-slate-400 animate-pulse">
            Agent 思考中...
          </div>
        )}
        <div ref={endRef} />
      </div>

      <div className="p-3 border-t border-slate-100 flex gap-2">
        <input
          className="flex-1 bg-slate-50 rounded-lg px-3 py-2 text-sm outline-none focus:ring-2 focus:ring-indigo-200"
          placeholder="输入消息..."
          value={input}
```

```tsx
"use client";

import { useEffect, useRef, useCallback, useState } from "react";
import { useBciEngine } from "@/hooks/use-bci-engine";
import { callAgent } from "@/lib/agent-api";
import { StatusCards } from "@/components/status-cards";
import { WaveformChart } from "@/components/waveform-chart";
import { BandBars } from "@/components/band-bars";
import { ChatPanel } from "@/components/chat-panel";
import type { ChatMessage } from "@/components/chat-panel";
import { StateSwitcher } from "@/components/state-switcher";
import type { BrainState } from "@/lib/signal-generator";

export default function Home() {
  const engine = useBciEngine();
  const lastStateRef = useRef("idle");
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [loading, setLoading] = useState(false);
  const msgIdRef = useRef(0);

  const addMessage = useCallback((type: ChatMessage["type"], text: string) => {
    const id = String(++msgIdRef.current);
    setMessages((prev) => [...prev, { id, type, text }]);
  }, []);

  const invokeAgent = useCallback(
    async (state: string, confidence: number, bands: Record<string, number>, message?: string) => {
      setLoading(true);
      try {
        const response = await callAgent({ state, confidence, bands, message });
        addMessage("agent", response);
      } catch (e) {
        addMessage("agent", `[错误] ${e instanceof Error ? e.message : "Agent 调用失败"}`);
      } finally {
        setLoading(false);
      }
    },
    [addMessage],
  );

  // Auto-invoke agent on state change
  useEffect(() => {
    const { state, confidence } = engine.decoded;
    if (state !== lastStateRef.current && state !== "idle" && confidence >= 0.5) {
      addMessage("system", `脑状态变化: ${lastStateRef.current} → ${state}`);
      invokeAgent(state, confidence, engine.bands);
      lastStateRef.current = state;
    }
  }, [engine.decoded, engine.bands, addMessage, invokeAgent]);

  // Auto-start
  useEffect(() => {
    engine.start();
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

  const handleSend = (text: string) => {
    addMessage("user", text);
    invokeAgent(engine.decoded.state, engine.decoded.confidence, engine.bands, text);
  };

  const handleSwitch = (state: BrainState) => {
    engine.switchState(state);
  };

  return (
    <div className="min-h-screen bg-[#f8f9fc]">
      {/* Nav */}
      <nav className="bg-white border-b border-slate-200 px-6 py-3 flex items-center justify-between">
        <div className="flex items-center gap-2">
          <div className="w-8 h-8 bg-indigo-500 rounded-lg flex items-center justify-center text-white text-sm font-bold">
            B
          </div>
          <span className="font-bold text-slate-800">BCI Agent</span>
          <span className="text-xs text-slate-400 ml-2">脑机接口 AI 智能体</span>
        </div>
        <StateSwitcher current={engine.decoded.state} onSwitch={handleSwitch} />
      </nav>

      {/* Main */}
      <div className="flex gap-4 p-4 h-[calc(100vh-57px)]">
        {/* Left: signal viz */}
        <div className="flex-[3] flex flex-col gap-3 min-w-0">
          <StatusCards
            state={engine.decoded.state}
            confidence={engine.decoded.confidence}
            isRunning={engine.isRunning}
          />
          <div className="flex-1">
            <WaveformChart data={engine.waveform} />
          </div>
          <BandBars bands={engine.bands} />
        </div>

        {/* Right: AI chat */}
        <div className="flex-[2] min-w-0">
          <ChatPanel messages={messages} onSend={handleSend} loading={loading} />
        </div>
      </div>
    </div>
  );
}
```
